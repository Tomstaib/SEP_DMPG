 65        """
 66        Create a custom composite tree with user input. A tree is symmetric since another form wouldn't make more sense.
 67        """
 68        try:
 69            while True:
 70
 71                num_children_per_parent: int = input_positive_number(
 72                    "Please enter how many children a Management Node should have")
 73                depth_of_tree: int = input_positive_number("Please enter the depth of the tree")
 74
 75                if cls.confirm_input(f"Do you want to proceed with these settings? [y/n]:"
 76                                     f"\nDepth of the tree: {depth_of_tree}"
 77                                     f"\nNumber of children per parent: {num_children_per_parent}"
 78                                     f"\nTotal number of compute Nodes: {num_children_per_parent ** depth_of_tree}"):
 79                    cls.destroy_composite_tree()
 80
 81                    current_level_nodes: list[ManagementNode] = [cls.__root]
 82
 83                    cls.__build_composite_tree(current_level_nodes, depth_of_tree, num_children_per_parent)
 84
 85                    break
 86                else:
 87                    if not cls.confirm_input("Do you want to create another composite tree? [y/n]: "):
 88                        print("Exiting the tree creation process.")
 89                        break

 92            print(f"Error: {str(e)}")

 94    @classmethod
 95    def __build_composite_tree(cls, current_level_nodes: list[Optional[ManagementNode]],
 96                               depth_of_tree: int, num_children_per_parent: int):
 97        """Build the composite tree based on provided inputs."""
 98        for depth in range(depth_of_tree):
 99            next_level_nodes: list = []
100            for parent_node in current_level_nodes:
101                for _ in range(num_children_per_parent):
102                    if depth == depth_of_tree - 1:
103                        child_node: ComputeNode = ComputeNode()
104                    else:
105                        child_node: ManagementNode = ManagementNode(parent=parent_node)
106                    parent_node.add(child_node)
107                    next_level_nodes.append(child_node)
108            current_level_nodes = next_level_nodes

273        """Helper method to find a node by its name."""
274        if current_node is None:
275            current_node: ManagementNode = cls.__root
276
277        if current_node.__str__() == name:
278            return current_node
279
280        if isinstance(current_node, ManagementNode):
281            for child in current_node:
282                result: Optional[Node] = cls._find_node_by_name(name, child)
283                if result is not None:
284                    return result

290        """Helper method to check if a node or its children have running ComputeNodes."""

302    def count_compute_nodes(cls, management_node: ManagementNode) -> int:
303        """
304        Counts the number of ComputeNodes under the given ManagementNode. There is a local function to do it
305        recursively.
306        """
307        count = 0
308
309        def _count_compute_nodes_recursively(node: ManagementNode):
310            """Recursively counts the number of ComputeNodes under the given ManagementNode."""
311            nonlocal count
312            for child in node:
313                if isinstance(child, ComputeNode):
314                    count += 1
315                elif isinstance(child, ManagementNode):
316                    _count_compute_nodes_recursively(child)
317
318        _count_compute_nodes_recursively(management_node)

377    def start_simulation(cls, num_replications: int, num_compute_nodes: int, slurm_account: str,
378                         slurm_username: str, model_script: str, time_limit: int, cpus_per_task: int):
379        """
380        Starts the simulation from the root node.
381
382        :param num_replications: The total number of replication to simulate.
383        :param num_compute_nodes: The number of compute nodes to simulate on.
384        :param slurm_account: The Slurm account to be billed.
385        :param slurm_username: The Slurm username to be used.
386        :param model_script: The model script to be used.
387        :param time_limit: The time limit for the Slurm job in minutes.
388        :param cpus_per_task: Number of CPUs per task for the simulation.
389        """
390        try:
391            total_compute_nodes: int = cls.count_compute_nodes(cls.__root)
392            not_running_compute_nodes: int = cls.count_not_running_compute_nodes(cls.__root)
393
394            print(f"Total ComputeNodes: {total_compute_nodes}")
395            print(f"ComputeNodes available for simulation (not running): {not_running_compute_nodes}")
396
397            if num_compute_nodes > not_running_compute_nodes:
398                num_compute_nodes: int = not_running_compute_nodes
399                print(f"Adjusting compute nodes to available number: {num_compute_nodes}")
400
401            if num_compute_nodes == 0:
402                print("No Compute Nodes available for computing")
403                return
404
405            replications_per_node: int = round(num_replications / num_compute_nodes)
406
407            print(f"Simulation will be started with {num_compute_nodes} Compute Nodes "
408                  f"and {replications_per_node} replications per node")
409
410            cls._run_simulation(cls.__root, num_compute_nodes, replications_per_node, slurm_account, slurm_username,
411                                model_script, time_limit, cpus_per_task)
412
413        except Exception as e:
414            print(f"Error starting simulation: {str(e)}")

429    @classmethod
430    def _run_simulation(cls, management_node: ManagementNode, num_compute_nodes_to_use: int, replications_per_node: int,
431                        slurm_account: str, slurm_username: str, model_script: str, time_limit: int,
432                        cpus_per_task: int) -> None:
433        """
434        Recursive method to start the simulation on the specified number of ComputeNodes with the given parameters.
435        """
436        count: int = 0
437
438        # Start monitoring the directory for job completion files
439        if cls.__observer is None:
440            cls.__start_file_monitoring(path_to_monitor=r"\util\flask\test")
441
442        def _simulate_on_node(node: Node):
443            nonlocal count
444            if count >= num_compute_nodes_to_use:
445                return
446
447            if isinstance(node, ComputeNode) and not node.is_running():
448                node.distribute_and_compute(model=model_script, num_replications=replications_per_node,
449                                            slurm_account=slurm_account, model_script=model_script,
450                                            time_limit=time_limit, slurm_username=slurm_username,
451                                            cpus_per_task=cpus_per_task)
452                count += 1
453
454            if isinstance(node, ManagementNode):
455                for child in node:
456                    _simulate_on_node(child)
457
458        _simulate_on_node(management_node)
459        print(f"Simulation started on {count} ComputeNodes.")
