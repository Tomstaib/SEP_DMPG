 28def run_simulation(model: Callable, minutes: Union[int, float], warm_up: Union[int, float] = None,
 29                   store_pivot_in_file: str = None) -> pd.DataFrame:
 30    """
 31    Run a simulation using the specified model for the given number of minutes.
 32
 33    :param model (Callable): The simulation model function.
 34    :param minutes (int): The number of minutes to run the simulation.
 35
 36    :return pivot_table (DataFrame): The pivot
 37    """
 38
 39    if warm_up is not None:
 40        set_duration_warm_up(warm_up)
 41
 42    print(gi.DURATION_WARM_UP)
 43
 44    random.seed(RANDOM_SEED)
 45    env = simpy.Environment()
 46    model(env)
 47    env.run(until=minutes)
 48
 49    # Get the statistics
 50    entity_stats, server_stats, sink_stats, source_stats = calculate_statistics(env)
 51
 52    # Structure the data for the DataFrame
 53    data = []
 54
 55    # Function to round the values
 56
 57    # Entity Stats
 58    for key, value in entity_stats.items():
 59        data.append({'Type': 'Entity', 'Name': 'Entity', 'Stat': key, 'Value': round_value(value)})
 60
 61    # Server Stats
 62    for stat in server_stats:
 63        for key, value in stat.items():
 64            if key != 'Server':
 65                data.append({'Type': 'Server', 'Name': stat['Server'], 'Stat': key, 'Value': round_value(value)})
 66
 67    # Sink Stats
 68    for sink_name, stats in sink_stats.items():
 69        for key, value in stats.items():
 70            data.append({'Type': 'Sink', 'Name': sink_name, 'Stat': key, 'Value': round_value(value)})
 71
 72    # Source Stats
 73    for source_name, stats in source_stats.items():
 74        for key, value in stats.items():
 75            data.append({'Type': 'Source', 'Name': source_name, 'Stat': key, 'Value': round_value(value)})
 76
 77    # Create DataFrame
 78    df = pd.DataFrame(data)
 79
 80    # Optimize DataFrame memory usage
 81    df['Value'] = pd.to_numeric(df['Value'], downcast='float')  # Downcast to float32 if possible
 82    df['Type'] = df['Type'].astype('category')  # Convert to categorical type
 83    df['Name'] = df['Name'].astype('category')  # Convert to categorical type
 84    df['Stat'] = df['Stat'].astype('category')  # Convert to categorical type
 85
 86    # Create Pivot Table
 87    pivot_table = df.pivot_table(index=['Type', 'Name', 'Stat'], values='Value', aggfunc='mean')
 88
 89    # Print Pivot Table
 90    logging.info(pivot_table)
 91
 92    # Optionally save to CSV
 93    if store_pivot_in_file:
 94        pivot_table.to_csv(store_pivot_in_file)
 95
 96    return pivot_table

 99def calculate_statistics(env) -> Tuple:
100    """
101    Calculate various statistics based on the simulation environment.
102
103    :param env: The simulation environment.
104
105    :return Tuple[Dict, List[Dict], Dict, Dict]: A tuple containing dictionaries for entity, server, sink,
106     and source statistics.
107    """
108    # Calculate entity statistics
109    all_entities_with_warm_up_time = EntityManager.entities
110    all_entities = []
111
112    for objekt in all_entities_with_warm_up_time:
113        if objekt.destruction_time is not None:
114            if objekt.destruction_time > gi.DURATION_WARM_UP:
115                all_entities.append(objekt)
116            else:
117                pass
118        else:
119            all_entities.append(objekt)
120
121    times_in_system = [entity.destruction_time - entity.creation_time
122                       for entity in all_entities if entity.destruction_time]
123
124    logging.debug(f"Entities not processed: {len(all_entities)} - {len(times_in_system)} "
125                  f"= {len(all_entities) - len(times_in_system)} "
126                  f"({len(times_in_system) / len(all_entities)} %)")
127
128    if times_in_system:
129        avg_time_in_system_pivot_table = sum(times_in_system) / len(times_in_system)
130        max_time_in_system_pivot_table = max(times_in_system)
131        min_time_in_system_pivot_table = min(times_in_system)
132    else:
133        avg_time_in_system_pivot_table = max_time_in_system_pivot_table = min_time_in_system_pivot_table = 0
134
135    number_created_pivot_table = len(all_entities)
136    number_destroyed_pivot_table = sum(entity.destruction_time is not None for entity in all_entities)
137
138    entity_stats = {
139        'NumberInSystem': len(all_entities),
140        'AvgTimeInSystem': avg_time_in_system_pivot_table,
141        'MaxTimeInSystem': max_time_in_system_pivot_table,
142        'MinTimeInSystem': min_time_in_system_pivot_table,
143        'NumberCreated': number_created_pivot_table,
144        'NumberDestroyed': number_destroyed_pivot_table
145    }
146    # Calculate server statistics
147    server_stats = []
148    for server in Server.servers:
149        current_simulation_time = env.now
150        scheduled_utilization_pivot_table = 0
151        avg_time_processing_pivot_table = 0
152        if env.now > gi.DURATION_WARM_UP:
153            scheduled_utilization_pivot_table = (server.units_utilized_pivot_table / current_simulation_time) * 100 \
154                if current_simulation_time > 0 else 0
155            avg_time_processing_pivot_table = (
156                server.total_processing_time_pivot_table / server.entities_processed
157                if server.entities_processed > 0 else 0)
158
159        server_stats.append({
160            'Server': server.name,
161            'ScheduledUtilization': scheduled_utilization_pivot_table,
162            'UnitsUtilized': server.units_utilized_pivot_table,
163            'AvgTimeProcessing': avg_time_processing_pivot_table,
164            'TotalTimeProcessing': server.total_processing_time_pivot_table,
165            'NumberEntered': server.number_entered_pivot_table,
166            'NumberExited': server.number_exited_pivot_table,
167            'NumberDowntimes': server.number_downtimes_pivot_table,
168            'TotalDowntime': server.total_downtime_pivot_table
169        })
170
171    # Calculate sink statistics
172    sink_stats = {}
173    for sink in Sink.sinks:
174
175        if sink.tally_statistic.num_times_processed_list:
176            tally_min, tally_max, tally_avg = sink.tally_statistic.calculate_statistics()
177        else:
178            tally_min, tally_max, tally_avg = None, None, None
179
180        avg_time_in_system_pivot_table = (
181            sink.total_time_in_system / sink.entities_processed
182            if sink.entities_processed > 0 else 0)
183
184        sink_stats[sink.name] = {
185            'AvgTimeInSystem': avg_time_in_system_pivot_table,
186            'MaxTimeInSystem': sink.max_time_in_system_pivot_table,
187            'MinTimeInSystem': sink.min_time_in_system_pivot_table if sink.entities_processed > 0 else None,
188            'NumberEntered': sink.number_entered_pivot_table,
189
190            'NumTimesProcessed_Avg': tally_avg,
191            'NumTimesProcessed_Max': tally_max,
192            'NumTimesProcessed_Min': tally_min,
193        }
194
195    # Calculate source statistics
196    source_stats = {}
197    for source in Source.sources:
198        source_stats[source.name] = {
199            'NumberCreated': source.entities_created_pivot_table,
200            'NumberExited': source.number_exited_pivot_table,
201        }
202
203    return entity_stats, server_stats, sink_stats, source_stats

206def replication(env_setup_func, calculate_stats_func, minutes, r) -> pd.DataFrame:
207    """
208    Replicate a simulation run.
209
210    :param env_setup_func (Callable): A function that sets up the simulation environment.
211    :param calculate_stats_func (Callable): A function that calculates statistics based on the simulation environment.
212    :param minutes (int): The number of minutes to run the replication.
213    :param r (int): iteration.
214
215    :return Tuple[Dict, List[Dict], Dict, Dict]: A tuple containing dictionaries for entity, server, sink, and source statistics.
216    """
217    random.seed(r)
218    EntityManager.destroy_all_entities()
219    Source.sources.reset_all()
220    Server.servers.reset_all()
221    Sink.sinks.reset_all()
222    env = simpy.Environment()
223    env_setup_func(env)
224    env.run(until=minutes)
225
226    result = calculate_stats_func(env)
227
228    del env
229    gc.collect()
230    return result

233def get_percentage_and_computingtimes(computing_time_start, i, num_replications) -> Tuple[str, str, str, str, str]:
234    """
235    Calculate the percentage completion and various time metrics for a set of replications.
236
237    param: computing_time_start (float): The start time of the computation.
238    param: i (int): The current iteration number.
239    param: num_replications (int): The total number of replications.
240
241    return: Tuple[Dict, List[Dict], Dict, Dict]: A tuple containing formatted strings for percentage completion and time metrics.
242    """
243    global seconds_previous_computations
244
245    seconds_computed = time.time() - computing_time_start
246    seconds_computed_iteration = seconds_computed - seconds_previous_computations
247    seconds_previous_computations = seconds_computed
248
249    percentage = round((i + 1) / num_replications * 100)
250    total_seconds_to_complete = seconds_computed / (percentage / 100)
251
252    return (f"{percentage:>3}%",
253            f"[time computed] {str(timedelta(seconds=seconds_computed)):<15}",
254            f"[time to complete] {str(timedelta(seconds=total_seconds_to_complete - seconds_computed)):<15}",
255            f"[time prediction] {str(timedelta(seconds=total_seconds_to_complete)):<15}",
256            f"[time per iteration] {str(timedelta(seconds=seconds_computed_iteration)):<15}")

394def create_pivot(all_entity_stats, all_server_stats, all_sink_stats, all_source_stats, entity_stat_names,
395                 server_stat_names, sink_stat_names, source_stat_names, store_pivot_in_file: str = None) -> tuple:
396    """
397        Create a pivot table from collected simulation statistics.
398
399        param: all_entity_stats (list): List of entity statistics from multiple replications.
400        param: all_server_stats (dict): Dictionary of server statistics from multiple replications.
401        param: all_sink_stats (dict): Dictionary of sink statistics from multiple replications.
402        param: all_source_stats (dict): Dictionary of source statistics from multiple replications.
403        param: entity_stat_names (list): List of entity statistics names.
404        param: server_stat_names (list): List of server statistics names.
405        param: sink_stat_names (list): List of sink statistics names.
406        param: source_stat_names (list): List of source statistics names.
407        """
408
409    def calculate_aggregate_stats(values) -> tuple:
410        """
411        Calculate aggregate statistics (average, minimum, maximum, half-width).
412
413        param: values (list): List of numeric values.
414
415        return: Tuple: Tuple containing average, minimum, maximum, half
416        """
417        numeric_values = [value for value in values if isinstance(value, (int, float))]
418        if not numeric_values:
419            return None, None, None, None
420
421        avg = np.mean(numeric_values)
422        min_val = np.min(numeric_values)
423        max_val = np.max(numeric_values)
424        half_width = 1.96 * (np.std(numeric_values) / np.sqrt(len(numeric_values)))
425        return avg, min_val, max_val, half_width
426
427    def flatten_stats(stats, name, stat_names, is_entity=False) -> list:
428        """
429        Flatten statistics into a list for DataFrame.
430
431        param: stats (dict): Dictionary containing statistics for a particular component.
432        param: name (str): Name of the component type (Entity, Server, Sink, Source).
433        param: stat_names (list): List of statistic names.
434        param: is_entity (bool): Indicates if the component is an entity.
435
436        return: List of dictionaries containing statistics for a particular component
437        """
438        flattened = []
439        for component_name, stat_values in stats.items():
440            for stat_name in stat_names:
441                values = stat_values.get(stat_name, {}) if is_entity else stat_values.get(stat_name)
442                if values:
443                    avg, min_val, max_val, half_width = values if is_entity else \
444                        (values[0], values[1], values[2], values[3])
445                    row = {'Type': name, 'Name': component_name, 'Stat': stat_name,
446                           'Average': round(avg, 4) if avg is not None else None,
447                           'Minimum': round(min_val, 4) if min_val is not None else None,
448                           'Maximum': round(max_val, 4) if max_val is not None else None,
449                           'Half-Width': round(half_width, 4) if half_width is not None else None}
450                    flattened.append(row)
451        return flattened
452
453    # Calculate aggregate statistics
454    entity_aggregate_stats = {stat: calculate_aggregate_stats([run[stat] for run in all_entity_stats])
455                              for stat in entity_stat_names}
456    modified_entity_stats = {'Entity': {}}
457    for stat_name, values in entity_aggregate_stats.items():
458        avg, min_val, max_val, half_width = values if values else (None, None, None, None)
459        modified_entity_stats['Entity'][stat_name] = (avg, min_val, max_val, half_width)
460    aggregate_server_stats = {server_name: {key: calculate_aggregate_stats([stat[key] for stat in stats_list])
461                                            for key in server_stat_names}
462                              for server_name, stats_list in all_server_stats.items()}
463    aggregate_sink_stats = {sink_name: {key: calculate_aggregate_stats([stat[key] for stat in stats_list])
464                                        for key in sink_stat_names}
465                            for sink_name, stats_list in all_sink_stats.items()}
466    aggregate_source_stats = {source_name: {key: calculate_aggregate_stats([stat[key] for stat in stats_list])
467                                            for key in source_stat_names}
468                              for source_name, stats_list in all_source_stats.items()}
469
470    # Flatten all stats into a single list
471    flattened_stats = []
472    flattened_stats.extend(flatten_stats(modified_entity_stats, 'Entity', entity_stat_names, is_entity=True))
473    flattened_stats.extend(flatten_stats(aggregate_server_stats, 'Server', server_stat_names))
474    flattened_stats.extend(flatten_stats(aggregate_sink_stats, 'Sink', sink_stat_names))
475    flattened_stats.extend(flatten_stats(aggregate_source_stats, 'Source', source_stat_names))
476
477    # Creating a combined DataFrame from flattened stats
478    df_combined = pd.DataFrame(flattened_stats)
479
480    pivot_table_combined = df_combined.pivot_table(
481        index=['Type', 'Name', 'Stat'],
482        values=['Average', 'Minimum', 'Maximum', 'Half-Width'],
483        aggfunc='mean'
484    )
485    # Reorder the columns
486    pivot_table_combined = pivot_table_combined[['Average', 'Minimum', 'Maximum', 'Half-Width']]
487    # Print the Pivot Table
488    logging.info("\n" + str(pivot_table_combined))
489
490    if store_pivot_in_file:
491        pivot_table_combined.to_csv('combined_simulation_stats.csv')
492
493    return pivot_table_combined
